31 template <void (Urho3D::RaycastVehicle::*mptr)(int, float)> class WheelServo : public Servo {
32  URHO3D_OBJECT(WheelServo, Servo);
33  static constexpr auto err_msg = "Attempted to create component ServoMotor with an invalid configuration";
34 
35  Urho3D::RaycastVehicle* vehicle;
36  std::vector<unsigned int> wheels;
37  float max_speed;
38 
39  public:
40  explicit WheelServo(BoardData& bd, Urho3D::Node* node, const rapidjson::Value& pin)
41  : Servo{bd, node, pin}, vehicle{node->GetComponent<Urho3D::RaycastVehicle>(true)}, wheels{} {
42  if (!pin.HasMember("wheels"))
43  throw std::runtime_error{err_msg};
44  if (!pin.HasMember("max_speed"))
45  throw std::runtime_error{err_msg};
46  pwm_pin->store(90);
47  max_speed = pin["max_speed"].GetFloat();
48  if (vehicle)
49  for (const auto& x : pin["wheels"].GetArray()) {
50  auto val = x.GetUint();
51  if (val > vehicle->GetNumWheels())
52  throw std::runtime_error{err_msg};
53  wheels.push_back(val);
54  }
55  }
56 
57  void Update(float timeStep) override {
58  if constexpr (to_enum<mptr>() == RayMembers::SetSteeringValue) {
59  for (const auto& no : wheels)
60  (vehicle->*mptr)(no, deg2rad(math_map(+pwm_pin->load(), 0, 180, movement_mode->angle_min, movement_mode->angle_max)));
61  } else if constexpr (to_enum<mptr>() == RayMembers::SetEngineForce) {
62  for (const auto& no : wheels)
63  (vehicle->*mptr)(no, math_map(static_cast<float>(+pwm_pin->load()), 0.f, 180.f, -1.f, +1.f) * max_speed);
64  }
65  }
66 };
