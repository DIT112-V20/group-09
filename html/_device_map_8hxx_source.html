69  constexpr static auto make_ticker(const DeviceMap& devmap, DeviceStorage& storage, I2cBus& bus, Driver& driver) {
70  return [&]() {
71  std::scoped_lock lock{bus.devices_mut};
72  auto& buf = bus.slaves[storage.address].first;
73  std::scoped_lock tx_lock{buf.tx_mutex};
74  std::size_t index = 0;
75  std::size_t size = buf.tx.size();
76  for (const auto& packet : buf.tx) {
77  if (packet.size() == 1) // Read byte written but handle operation did not lock yet
78  break; // Let handle operation lock
79  const auto reg = static_cast<uint8_t>(packet.front());
80  switch (devmap.map[reg]) {
81  case MapVals::store:
82  std::memcpy(&storage.data[reg], packet.data() + 1, packet.size() - 1);
83  break;
84  case MapVals::invoke:
85  devmap.func[reg](driver, storage, gsl::make_span(packet.data() + 1, packet.size() - 1));
86  }
87  ++index;
88  }
89 
90  if (index == buf.tx.size())
91  buf.tx = {};
92  else
93  buf.tx.erase(buf.tx.begin(), buf.tx.begin() + index);

95  }
