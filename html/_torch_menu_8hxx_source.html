91  using handle_func = void (TorchMenu::*)(Urho3D::StringHash, Urho3D::VariantMap& event_data);
92  template <handle_func CancelHandler, handle_func OkHandler> void handle_get_path(Urho3D::StringHash, Urho3D::VariantMap& event_data) {
93  auto* el = static_cast<Urho3D::UIElement*>(event_data[Urho3D::Released::P_ELEMENT].GetPtr());
94  if (!selector_data_map.contains(el))
95  return;
96 
97  auto& dat = selector_data_map[el];
98 
99  selector = Urho3D::MakeShared<Urho3D::FileSelector>(context_);
100  selector->GetWindow()->SetPriority(999);
101  selector->SetDefaultStyle(styling);
102  selector->SetButtonTexts(dat.ok_text.c_str(), dat.cancel_text.c_str());
103  selector->SetTitle(dat.window_name.c_str());
104 
105  auto filter = Urho3D::Vector<Urho3D::String>{};
106  ranges::for_each(dat.file_filters, [&](const auto& x) { filter.Push(x.c_str()); });
107  selector->SetFilters(filter, 0);
108 
109  SubscribeToEvent(selector->GetCancelButton(), Urho3D::E_RELEASED, new Urho3D::EventHandlerImpl<TorchMenu>(this, CancelHandler));
110  SubscribeToEvent(selector->GetOKButton(), Urho3D::E_RELEASED, new Urho3D::EventHandlerImpl<TorchMenu>(this, OkHandler));
111  }
